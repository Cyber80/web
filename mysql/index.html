1) รู้จักฐานข้อมูลและ MySQL

ฐานข้อมูล (Database) คือที่เก็บข้อมูลจำนวนมากอย่างเป็นระบบ เพื่อค้นหาและปรับปรุงได้สะดวก
  • ตาราง (Table) = เหมือนชีตในสเปรดชีต
  • แถว/ระเบียน (Row/Record) = ข้อมูล 1 รายการ
  • คอลัมน์/ฟิลด์ (Column/Field) = ชนิดข้อมูลแต่ละช่อง
  • กุญแจหลัก (Primary Key) = รหัสไม่ซ้ำ เช่น student_id
MySQL คือระบบฐานข้อมูลแบบสัมพันธ์ (Relational DBMS) ที่ใช้ภาษา SQL ในการสั่งงาน
หมายเหตุ: SQL เป็นมาตรฐาน ส่วน MySQL มีไวยากรณ์เฉพาะบางส่วน แต่คำสั่งพื้นฐานใช้ได้ร่วมกันส่วนใหญ่

3) โครงสร้างคำสั่ง SQL เบื้องต้น

รูปแบบความหมาย
CREATE TABLE ...สร้างตารางใหม่
INSERT INTO ...เพิ่มข้อมูล
SELECT ... FROM ...อ่านข้อมูล
UPDATE ... SET ...แก้ไขข้อมูล
DELETE FROM ...ลบข้อมูล
เคล็ดลับ: ตัวพิมพ์เล็ก-ใหญ่ไม่สำคัญสำหรับคำสั่ง แต่ควรเขียนคำสงวนเป็นตัวใหญ่เพื่ออ่านง่าย เช่น SELECT, FROM, WHERE

4) ชุดข้อมูลตัวอย่าง: โรงเรียน

ใช้ฐานข้อมูล school_db ที่มี 3 ตาราง: students, classes, scores

สร้างตาราง

DDL (สร้างโครงสร้าง) 
CREATE TABLE classes (
  class_id INT PRIMARY KEY AUTO_INCREMENT,
  class_name VARCHAR(50) NOT NULL,
  teacher VARCHAR(60) NOT NULL
);
CREATE TABLE students (
  student_id INT PRIMARY KEY AUTO_INCREMENT,
  full_name VARCHAR(80) NOT NULL,
  gender ENUM('M','F') NOT NULL,
  birthdate DATE NOT NULL,
  class_id INT,
  CONSTRAINT fk_class FOREIGN KEY (class_id) REFERENCES classes(class_id)
);
CREATE TABLE scores (
  score_id INT PRIMARY KEY AUTO_INCREMENT,
  student_id INT NOT NULL,
  subject VARCHAR(40) NOT NULL,
  term VARCHAR(10) NOT NULL,
  score DECIMAL(5,2) NOT NULL CHECK (score BETWEEN 0 AND 100),
  CONSTRAINT fk_student FOREIGN KEY (student_id) REFERENCES students(student_id)
);

5) CRUD: สร้าง อ่าน แก้ไข ลบ

SELECT (อ่าน)

เลือกคอลัมน์ที่ต้องการ 
SELECT student_id, full_name FROM students;
นับจำนวนแถว 
SELECT COUNT(*) AS total_students FROM students;

INSERT / UPDATE / DELETE

INSERT เพิ่มนักเรียน 
INSERT INTO students (full_name, gender, birthdate, class_id)
VALUES ('ปุณณวิชญ์ ก้อง','M','2007-02-14',2);
UPDATE เปลี่ยนห้อง 
UPDATE students SET class_id = 3 WHERE student_id = 1;
DELETE ลบคะแนนผิดพลาด 
DELETE FROM scores WHERE score > 100;

6) คัดกรอง จัดเรียง จำกัดผล

WHERE

คณิตศาสตร์มากกว่า 80 
SELECT * FROM scores WHERE subject = 'คณิตศาสตร์' AND score > 80;
ระหว่าง 70-85 
SELECT * FROM scores WHERE score BETWEEN 70 AND 85;

ORDER BY / LIMIT

เรียงจากมากไปน้อย 
SELECT full_name, birthdate FROM students ORDER BY birthdate ASC;
เอา 3 แถวแรก 
SELECT * FROM scores ORDER BY score DESC LIMIT 3;

7) การ JOIN ตาราง

JOIN ใช้เชื่อมข้อมูลจากหลายตารางผ่านคีย์ที่สัมพันธ์กัน

รายชื่อนักเรียนพร้อมชื่อห้อง 
SELECT s.student_id, s.full_name, c.class_name
FROM students s
JOIN classes c ON s.class_id = c.class_id;
คะแนนพร้อมชื่อ นร. และห้อง 
SELECT s.full_name, c.class_name, sc.subject, sc.term, sc.score
FROM scores sc
JOIN students s ON sc.student_id = s.student_id
LEFT JOIN classes c ON s.class_id = c.class_id;

8) สรุปผลด้วย Aggregates และ GROUP BY

ค่าเฉลี่ย/สูงสุด/ต่ำสุด

เฉลี่ยคะแนนคณิตฯ รายชั้น 
SELECT c.class_name, AVG(sc.score) AS avg_math
FROM scores sc
JOIN students s ON sc.student_id = s.student_id
JOIN classes c ON s.class_id = c.class_id
WHERE sc.subject = 'คณิตศาสตร์'
GROUP BY c.class_name
ORDER BY avg_math DESC;

HAVING (กรองหลังสรุปแล้ว)

เฉลี่ย >= 80 
SELECT subject, AVG(score) AS avg_score
FROM scores
GROUP BY subject
HAVING AVG(score) >= 80;

9) ฟังก์ชันที่พบบ่อย

สตริง

SELECT UPPER(full_name) AS upper_name FROM students;
SELECT SUBSTRING(full_name,1,3) FROM students;

ตัวเลข

SELECT ROUND(AVG(score),2) FROM scores;
SELECT CEIL(78.2), FLOOR(78.9);

วันที่/เวลา

SELECT TIMESTAMPDIFF(YEAR, birthdate, CURDATE()) AS age FROM students;
SELECT DATE_FORMAT(birthdate, '%d/%m/%Y') FROM students;

10) Subquery (คำสั่งซ้อน)

หานักเรียนที่คะแนนคณิตฯ สูงกว่าค่าเฉลี่ยทั้งโรงเรียน 
SELECT s.full_name, sc.score
FROM scores sc
JOIN students s ON sc.student_id = s.student_id
WHERE sc.subject = 'คณิตศาสตร์'
AND sc.score > (
  SELECT AVG(score) FROM scores WHERE subject = 'คณิตศาสตร์'
);

11) ดัชนี (Index)

ดัชนีช่วยให้ค้นหาข้อมูลเร็วขึ้น โดยเฉพาะคอลัมน์ที่ถูกใช้ในเงื่อนไข WHERE หรือการเชื่อมตาราง

สร้างดัชนีที่ subject+term 
CREATE INDEX idx_scores_subject_term ON scores(subject, term);

ข้อควรระวัง: ดัชนีช่วยอ่านข้อมูลเร็วขึ้น แต่ทำให้การ INSERT/UPDATE ช้าลงเล็กน้อย

12) การออกแบบฐานข้อมูลเบื้องต้น (Normalization ย่อ)

แนวคิด

  • ข้อมูลไม่ซ้ำซ้อน
  • แบ่งตารางตามหน่วยข้อมูล (เช่น นักเรียน/ห้อง/คะแนน)
  • ใช้กุญแจหลัก (PK) และกุญแจนอก (FK) ให้สัมพันธ์กัน

ตัวอย่างออกแบบ

students(student_id PK, full_name, gender, birthdate, class_id FK)
classes(class_id PK, class_name, teacher)
scores(score_id PK, student_id FK, subject, term, score)

13) แบบฝึกหัด

  1. หาชื่อนักเรียนทุกคนในห้อง ม.5/1
  2. หาคะแนนเฉลี่ยวิชา ภาษาไทย ของเทอม 1/2567
  3. หานักเรียนที่อายุมากกว่า 17 ปี
  4. สรุปค่าเฉลี่ยคณิตฯ แยกตามห้อง และเรียงจากมากไปน้อย
  5. เพิ่มนักเรียนใหม่ 1 คน และเพิ่มคะแนนอย่างน้อย 2 วิชา
เฉลยตัวอย่างสั้น ๆ 
-- 1
SELECT s.full_name
FROM students s
JOIN classes c ON s.class_id = c.class_id
WHERE c.class_name = 'ม.5/1';

-- 2
SELECT AVG(score) AS avg_th
FROM scores
WHERE subject='ภาษาไทย' AND term='1/2567';

-- 3
SELECT full_name
FROM students
WHERE TIMESTAMPDIFF(YEAR, birthdate, CURDATE()) > 17;

14) ชีตสรุป (Cheat Sheet)

-- เลือกคอลัมน์
SELECT col1, col2 FROM t;

-- เงื่อนไข
SELECT * FROM t WHERE col > 10;
-- จัดเรียง/จำกัด
SELECT * FROM t
ORDER BY col DESC
LIMIT 5;
-- สรุปผล
SELECT col, AVG(x)
FROM t
GROUP BY col;
