SQL Syntax

ไวยากรณ์ SQL คือชุดของกฎและคำสั่งที่ใช้ในการสื่อสารกับฐานข้อมูล...

คำสั่ง SQL ส่วนใหญ่จะเริ่มต้นด้วยคำกริยา เช่น SELECT, UPDATE, DELETE, INSERT INTO, CREATE DATABASE, ALTER DATABASE, CREATE TABLE, ALTER TABLE, DROP TABLE, CREATE INDEX, DROP INDEX.

คำสั่ง SELECT:

ใช้สำหรับเลือกข้อมูลจากฐานข้อมูล

                
                    SELECT column1, column2, ...
                    FROM table_name;

หรือเลือกข้อมูลทั้งหมด:

                
                    SELECT *
                    FROM table_name;

คำสั่ง UPDATE:

ใช้สำหรับแก้ไขข้อมูลที่มีอยู่ในตาราง

                
                    UPDATE table_name
                    SET column1 = value1, column2 = value2, ...
                    WHERE condition;
